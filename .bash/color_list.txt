#!/usr/bin/env bash

get_branch="\$(git_branch)";
get_branch_length=${#get_branch};
#https://tldp.org/HOWTO/Bash-Prompt-HOWTO/x405.html
#https://wiki.archlinux.org/title/Bash/Prompt_customization

# Color
tput sgr0; # reset colors
bold=$(tput bold);
italic=$(tput sitm);
blink=$(tput blink);
inverted=$(tput rev);
underlined=$(tput smul);
#movement
pos_save=$(tput sc);

#	pos_right_branch=$(tput hpa $((COLUMNS - get_branch_length - 1)));

    pos_line_down=$(tput cud 1)
pos_line_up=$(tput cuu 1)
#removing
clear_left=$(tput el)
    reset=$(tput sgr0);
#especials
# http://heyrod.com/snippets/tag-bash.html
    wrap_off=$(tput rmam)
wrap_on=$(tput smam)
# Solarized colors, taken from http://git.io/solarized-colors.
    black=$(tput setaf 0);
    lgray=$(tput setaf 7);
    dgray=$(tput setaf 8);
    blue=$(tput setaf 27);
    cyan=$(tput setaf 39);
    green=$(tput setaf 34);
    orange=$(tput setaf 166);
    purple=$(tput setaf 127);
    red=$(tput setaf 124);
    violet=$(tput setaf 99);
    white=$(tput setaf 15);
    yellow=$(tput setaf 220);
    brow=$(tput setaf 52);
# Set a background color
    bblack=$(tput setab 0);
    blgrey=$(tput setab 7);
    bdgrey=$(tput setab 8);
    bblue=$(tput setab 27);
    bcyan=$(tput setab 39);
